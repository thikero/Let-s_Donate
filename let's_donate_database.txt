create table banks(
bank_id int(25) not null auto_increment,
bank_name varchar(50) not null,
bank_type varchar(25) not null,
bank_phone_number int(20) not null,
bank_discription text,
bank_location varchar(100),
bank_requirement text,
bank_phot blob,
bank_city varchar(25) not null,
primary key(bank_id)
);

create table user(
user_id int(11) not null auto_incrment,
user_name varchar(30) not null,
user_email varchar(25) not null,
user_password varchar(30) not null,
user_location varchar(75) not null,
user_blood_type varchar(10) not null,
user_dob date not null,
user_weight int(11),
user_photo blob,
primary key(user_id)
);

create table charity(
charity_id int(10) not null auto_increment,
charity_phone_no int(20) not null,
charity_name varchar(50) not null,
charity discription text,
charity_address varchar(70) not null,
charity_facebook varchar(100),
charity_photo blob,
primary key(charity_id)
);

create table posts(
post_id int(20) not null auto_increment,
post_type varchar(20) not null,
post_blood_type varchar(15) not null,
user_id int(20) not null,
post_phone_number int(15) not null,
post_emain varchar(30) not null,
post_description text,
primary key(post_id)
foreign key(user_id) references user(user_id)
);
